create database if not exists trip_article;
use trip_article;

-- Article table:
drop table if exists article_content;
create table article_content
(
id bigint not null auto_increment,
user_id bigint,
state char(2),
day_count tinyint,
content_head varchar(5000),
content_tail varchar(5000),
create_time datetime,
post_status boolean,
primary key (id)
);

drop table if exists article_day;
create table article_day
(
id bigint not null auto_increment,
article_id bigint,
day_id tinyint,
day_count tinyint,
content varchar(5000),
primary key (id)
);

drop table if exists article_route;
create table article_route
(
id bigint not null auto_increment,
article_id bigint,
day_id tinyint,
day_count tinyint,
dest_order tinyint,
destination_id bigint,
primary key (id)
);


-- Destination table:
create database if not exists trip_destination;
use trip_destination;

drop table if exists states;
create table states
(
id bigint not null auto_increment,
name char(2),
abbr varchar(600),
description varchar(64),
primary key (id)
);

drop table if exists destination;
create table destination
(
id bigint not null auto_increment,
attraction varchar(64),
state char(2),
detail varchar(600),
address varchar(64),
pic varchar(500),
primary key (id)
);

-- User table:
create database if not exists trip_user;
use trip_user;

drop table if exists user_info;
create table user_info
(
id bigint not null auto_increment,
username varchar(20),
password varchar(100),
primary key (id)
);


-- Comment table:
create database if not exists trip_comment;
use trip_comment;

drop table if exists article_comments;
create table article_comments
(
id bigint not null auto_increment,
user_id bigint,
article_id bigint,
content varchar(100),
create_time datetime,
primary key (id)
);

drop table if exists comment_comments;
create table comment_comments
(
id bigint not null auto_increment,
user_id bigint,
comment_id bigint,
content varchar(100),
create_time datetime,
primary key (id)
);

-- Statistic table:
create database if not exists trip_statistic;
use trip_comment;

drop table if exists user_collection;
create table user_collection
(
id bigint not null auto_increment,
user_id bigint,
article_id bigint,
primary key (id)
);

drop table if exists user_article_like;
create table user_article_like
(
id bigint not null auto_increment,
user_id bigint,
article_id bigint,
primary key (id)
);

drop table if exists user_comment_like;
create table user_comment_like
(
id bigint not null auto_increment,
user_id bigint,
comment_id bigint,
primary key (id)
);
